        // QUIZ ENGINE

        let currentQuizState = {

            currentQuestion: 0,

            score: 0,

            selectedAnswer: null,

            answers: []

        };

 

        function renderQuiz(quizData) {

            const quiz = quizData.quiz;

            const question = quiz.questions[currentQuizState.currentQuestion];

           

            return `

                <div class="max-w-3xl mx-auto">

                    <div class="mb-6">

                        <h2 class="text-2xl font-bold text-gray-800 mb-2">${quiz.title}</h2>

                        <div class="flex items-center justify-between">

                            <span class="text-gray-600">Question ${currentQuizState.currentQuestion + 1} of ${quiz.questions.length}</span>

                            <span class="bg-indigo-100 px-3 py-1 rounded-full text-indigo-800 font-semibold">Score: ${currentQuizState.score}/${quiz.questions.length}</span>

                        </div>

                        <div class="w-full bg-gray-200 rounded-full h-2 mt-3">

                            <div class="bg-indigo-600 h-2 rounded-full transition-all" style="width: ${((currentQuizState.currentQuestion + 1) / quiz.questions.length) * 100}%"></div>

                        </div>

                    </div>

 

                    <div class="bg-gray-50 rounded-lg p-6 mb-6">

                        <h3 class="text-xl font-semibold mb-6">${question.question}</h3>

                        <div class="space-y-3" id="quiz-options">

                            ${question.options.map((opt, idx) => `

                                <button onclick="selectAnswer(${idx})"

                                    class="quiz-option w-full text-left p-4 rounded-lg border-2 ${getOptionClass(idx, question.correct)}

                                    ${currentQuizState.selectedAnswer === null ? 'hover:border-indigo-400 cursor-pointer' : 'cursor-default'}"

                                    ${currentQuizState.selectedAnswer !== null ? 'disabled' : ''}>

                                    <span class="font-semibold">${String.fromCharCode(65 + idx)}.</span> ${opt}

                                    ${currentQuizState.selectedAnswer !== null && idx === question.correct ? ' ✓' : ''}

                                    ${currentQuizState.selectedAnswer === idx && idx !== question.correct ? ' ✗' : ''}

                                </button>

                            `).join('')}

                        </div>

                    </div>

 

                    ${currentQuizState.selectedAnswer !== null ? `

                        <div class="bg-${currentQuizState.selectedAnswer === question.correct ? 'green' : 'red'}-50 border-2 border-${currentQuizState.selectedAnswer === question.correct ? 'green' : 'red'}-200 rounded-lg p-6 mb-6">

                            <h4 class="font-bold text-lg mb-2 text-${currentQuizState.selectedAnswer === question.correct ? 'green' : 'red'}-800">

                                ${currentQuizState.selectedAnswer === question.correct ? '✓ Correct!' : '✗ Incorrect'}

                            </h4>

                            <p class="text-gray-800 whitespace-pre-line">${question.explanation}</p>

                        </div>

                        <button onclick="nextQuestion()"

                            class="w-full bg-indigo-600 text-white py-3 rounded-lg font-semibold hover:bg-indigo-700">

                            ${currentQuizState.currentQuestion < quiz.questions.length - 1 ? 'Next Question' : 'View Results'}

                        </button>

                    ` : ''}

                </div>

            `;

        }

 

        function getOptionClass(idx, correct) {

            if (currentQuizState.selectedAnswer === null) {

                return 'border-gray-300 bg-white';

            }

            if (idx === correct) {

                return 'border-green-500 bg-green-100';

            }

            if (idx === currentQuizState.selectedAnswer) {

                return 'border-red-500 bg-red-100';

            }

            return 'border-gray-300 bg-gray-100';

        }

 

        function selectAnswer(idx) {

            if (currentQuizState.selectedAnswer !== null) return;

           

            const quiz = getCurrentQuiz();

            const question = quiz.quiz.questions[currentQuizState.currentQuestion];

            currentQuizState.selectedAnswer = idx;

           

            if (idx === question.correct) {

                currentQuizState.score++;

            }

           

            currentQuizState.answers.push({

                question: currentQuizState.currentQuestion,

                correct: idx === question.correct

            });

           

            loadSection(getCurrentSection());

        }

 

        function nextQuestion() {

            const quiz = getCurrentQuiz();

            if (currentQuizState.currentQuestion < quiz.quiz.questions.length - 1) {

                currentQuizState.currentQuestion++;

                currentQuizState.selectedAnswer = null;

                loadSection(getCurrentSection());

            } else {

                showResults();

            }

        }

 

        function showResults() {

            const quiz = getCurrentQuiz();

            const percentage = Math.round((currentQuizState.score / quiz.quiz.questions.length) * 100);

           

            document.getElementById('content').innerHTML = `

                <div class="max-w-2xl mx-auto text-center">

                    <div class="text-6xl mb-4">🎉</div>

                    <h2 class="text-3xl font-bold mb-4">Quiz Complete!</h2>

                    <div class="text-5xl font-bold text-indigo-600 mb-2">${percentage}%</div>

                    <p class="text-xl text-gray-600 mb-6">You scored ${currentQuizState.score} out of ${quiz.quiz.questions.length}</p>

                   

                    <div class="bg-gray-50 rounded-lg p-6 mb-6">

                        <h3 class="font-semibold text-lg mb-3">Performance Breakdown</h3>

                        <div class="space-y-2">

                            ${currentQuizState.answers.map((ans, idx) => `

                                <div class="flex items-center justify-between">

                                    <span>Question ${idx + 1}</span>

                                    <span>${ans.correct ? '✓' : '✗'}</span>

                                </div>

                            `).join('')}

                        </div>

                    </div>

 

                    <button onclick="resetQuiz()"

                        class="w-full bg-indigo-600 text-white py-3 rounded-lg font-semibold hover:bg-indigo-700 mb-3">

                        Retake Quiz

                    </button>

                    <button onclick="loadSection('home')"

                        class="w-full bg-gray-600 text-white py-3 rounded-lg font-semibold hover:bg-gray-700">

                        Back to Home

                    </button>

                </div>

            `;

        }

 

        function resetQuiz() {

            currentQuizState = {

                currentQuestion: 0,

                score: 0,

                selectedAnswer: null,

                answers: []

            };

            loadSection(getCurrentSection());

        }

 

        // NAVIGATION

        let currentCourse = 'chem122';

        let currentSection = 'home';

 

        function getCurrentCourse() {

            return courseData.courses.find(c => c.id === currentCourse);

        }

 

        function getCurrentSection() {

            return currentSection;

        }

 

        function getCurrentQuiz() {

            const course = getCurrentCourse();

            return course.sections.find(s => s.id === currentSection);

        }

 

        function loadSection(sectionId) {

            currentSection = sectionId;

            const course = getCurrentCourse();

            const section = course.sections.find(s => s.id === sectionId);

           

            if (section.type === 'quiz') {

                document.getElementById('pageTitle').textContent = section.title;

                document.getElementById('pageSubtitle').textContent = 'Test your knowledge';

                document.getElementById('content').innerHTML = renderQuiz(section);

            } else {

                document.getElementById('pageTitle').textContent = section.content.title;

                document.getElementById('pageSubtitle').textContent = section.content.subtitle;

                document.getElementById('content').innerHTML = section.content.body;

            }

           

            renderSidebar();

        }

 

        function renderSidebar() {

            const course = getCurrentCourse();

            const sidebar = document.getElementById('sidebar');

           

            sidebar.innerHTML = course.sections.map(section => `

                <button onclick="loadSection('${section.id}')"

                    class="sidebar-link w-full text-left px-4 py-3 rounded-lg ${currentSection === section.id ? 'bg-indigo-100 text-indigo-700 font-semibold' : 'text-gray-700 hover:bg-gray-100'}">

                    ${section.title}

                </button>

            `).join('');

        }

 

        // Initialize

        loadSection('home');
